# Chat System

A fully-featured Twitch-style chat system with user login, points economy, superchat tiers, and simulated messages.

## Overview

The chat system includes:
- **User authentication** with customizable username and color
- **Points economy** - users earn 2 points every 30 seconds
- **Three-tier superchats** - Bronze (40pts), Silver (80pts), Gold (120pts)
- **Simulated messages** from moderators and users
- **Live viewer count** with randomized fluctuation
- **Emoji quick-select** buttons

### Points Economy

#### updatePointsUI()

Updates the points display and enables/disables tier buttons.

**Location:** `script.js:533`

```javascript
function updatePointsUI() {
  if (!currentUser) return;
  const points = Math.max(0, Math.floor(currentUser.points || 0));
  const selectedTier = getTierByKey(selectedSuperchatTier);

  // Update points badge
  if (chatPoints) {
    chatPoints.innerHTML = `<i class="fa-solid fa-star"></i> ${points} pts`;
  }

  // Enable/disable tier buttons based on points
  tierButtons.forEach((btn) => {
    const cost = Number(btn.dataset.cost || 0);
    btn.disabled = points < cost;
  });

  // Auto-downgrade tier if insufficient points
  if (points < selectedTier.amount) {
    const best = getTierData(points);
    setSelectedTier(best ? best.css.replace("tier-", "") : "bronze");
  }

  // Update superchat send button
  if (chatSuperchatSend) {
    chatSuperchatSend.classList.remove("tier-bronze", "tier-silver", "tier-gold");
    const activeTier = getTierByKey(selectedSuperchatTier);
    if (points < activeTier.amount) {
      chatSuperchatSend.disabled = true;
      chatSuperchatSend.title = `Te faltan ${SUPERCHAT_COST_POINTS - points} puntos`;
      chatSuperchatSend.textContent = `Superchat ${SUPERCHAT_COST_POINTS} pts`;
    } else {
      chatSuperchatSend.disabled = false;
      chatSuperchatSend.classList.add(activeTier.css);
      chatSuperchatSend.title = `Enviar superchat ${activeTier.label} (${activeTier.amount} pts)`;
      chatSuperchatSend.innerHTML = `<i class="fa-solid ${activeTier.icon}"></i> ${activeTier.label} ${activeTier.amount} pts`;
    }
  }
}
```

### Message System

#### addMessage()

Adds a regular chat message.

**Location:** `script.js:598`

```javascript
function addMessage(name, text, color, isMod) {
  if (!chatMessages) return;
  
  const div = document.createElement("div");
  div.className = "chat-msg";
  const badge = isMod ? `<i class="fa-solid fa-shield-halved chat-mod" title="Moderador"></i>` : "";
  div.innerHTML = `${badge}<span class="chat-name" style="color:${color}">${name}:</span> ${text}<span class="chat-time">${getTimeStr()}</span>`;
  
  chatMessages.appendChild(div);
  
  // Limit normal messages (keep welcome + 40 messages)
  while (chatMessages.querySelectorAll('.chat-msg').length > 40) {
    chatMessages.querySelector('.chat-msg')?.remove();
  }

  // Limit superchats (max 3)
  while (chatMessages.querySelectorAll('.chat-superchat').length > 3) {
    chatMessages.querySelector('.chat-superchat')?.remove();
  }
  
  chatMessages.scrollTop = chatMessages.scrollHeight;
}
```

**Limits:**
- Maximum 40 regular messages displayed
- Maximum 3 superchats displayed simultaneously
- Auto-scrolls to bottom

#### addSuperchat()

Adds a highlighted superchat message.

```javascript
function addSuperchat(name, text, amount = "40 pts", isMod = true) {
  if (!chatMessages) return;
  const points = parsePoints(amount);
  const tier = getTierData(points) || { css: "tier-bronze", label: "Bronce" };
  
  // Create superchat element
  const div = document.createElement("div");
  div.className = `chat-superchat ${tier.css}`;
  const roleBadge = isMod
    ? '<i class="fa-solid fa-shield-halved chat-mod" title="Moderador"></i>'
    : '<i class="fa-solid fa-user chat-mod" title="Miembro"></i>';

  div.innerHTML = `
    <div class="superchat-badge"><i class="fa-solid fa-star"></i> ${tier.label} SUPERCHAT ${amount}</div>
    <div><span class="chat-name">${name}</span> ${roleBadge}</div>
    <div class="superchat-text">${text}</div>
    <span class="chat-time">${getTimeStr()}</span>
  `;
  
  chatMessages.appendChild(div);
  
  // Limit superchats
  while (chatMessages.querySelectorAll('.chat-superchat').length > 3) {
    chatMessages.querySelector('.chat-superchat')?.remove();
  }
  
  chatMessages.scrollTop = chatMessages.scrollHeight;
  
  // Update banner strip
  if (superchatBanner) {
    highlightedSuperchats.unshift({ name, amount });
    while (highlightedSuperchats.length > 2) highlightedSuperchats.pop();
    refreshSuperchatStrip();
  }
}
```

**Features:**
- Automatic tier detection based on points
- Adds to top of superchat banner (max 2)
- Different badge for moderators vs members

### Tier System

#### getTierData()

Returns tier configuration based on point amount.

```javascript
function getTierData(points) {
  if (points >= 120) return { css: "tier-gold", icon: "fa-crown", label: "Oro", amount: 120 };
  if (points >= 80) return { css: "tier-silver", icon: "fa-gem", label: "Plata", amount: 80 };
  if (points >= 40) return { css: "tier-bronze", icon: "fa-medal", label: "Bronce", amount: 40 };
  return null;
}
```

```javascript
function sendUserSuperchat() {
  if (!currentUser || !chatInput) return;

  const text = chatInput.value.trim();
  const tier = getTierByKey(selectedSuperchatTier);
  const amount = tier.amount;
  if (!text) {
    chatInput.focus();
    return;
  }

  // Verify points
  if ((currentUser.points || 0) < amount) {
    addMessage("Sistema", `No tienes puntos suficientes para un superchat de ${amount} pts.`, "rgba(255,255,255,.5)", false);
    return;
  }

  // Deduct points
  currentUser.points -= amount;
  saveCurrentUser();
  updatePointsUI();

  addSuperchat(currentUser.name, text, `${amount} pts`, false);
  chatInput.value = "";
  chatInput.focus();
}
```
