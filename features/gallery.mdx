# Masonry Gallery

An interactive photo gallery with CSS masonry layout, full-screen lightbox viewer, and vintage paper frame styling.

## Overview

The gallery system includes:
- **CSS Grid masonry layout** for dynamic photo arrangement
- **Lightbox modal** with full-screen image viewing
- **Paper frame effect** with rotated photo styling
- **Hover overlays** with captions
- **Keyboard navigation** (Escape to close)
- **Lazy loading** for performance

## Core Function

### initLightbox()

Initializes the lightbox modal system for full-screen image viewing.

**Location:** `script.js:326`

```javascript
function initLightbox() {
  const galleryGrid = $("#galleryGrid");
  const lightbox = $("#lightbox");
  const lightboxImg = $("#lightboxImg");
  const lightboxCaption = $("#lightboxCaption");
  const lightboxClose = $("#lightboxClose");

  if (!galleryGrid || !lightbox || !lightboxImg) return;

  // Open lightbox on image click
  galleryGrid.addEventListener("click", (e) => {
    const shot = e.target.closest(".shot");
    if (!shot) return;
    
    const img = shot.querySelector("img");
    if (!img) return;

    lightboxImg.src = img.src;
    lightboxImg.alt = img.alt;
    
    const caption = shot.dataset.caption || img.alt;
    if (lightboxCaption) lightboxCaption.textContent = caption;
    
    lightbox.classList.add("open");
    document.body.style.overflow = "hidden";
  });

  function closeLightbox() {
    lightbox.classList.remove("open");
    document.body.style.overflow = "";
  }

  // Close button
  lightboxClose?.addEventListener("click", closeLightbox);
  
  // Click outside image
  lightbox?.addEventListener("click", (e) => {
    if (e.target === lightbox) closeLightbox();
  });
  
  // Escape key
  document.addEventListener("keydown", (e) => {
    if (e.key === "Escape") closeLightbox();
  });
}
```

### Event Delegation

The lightbox uses **event delegation** on the gallery grid:

```javascript
galleryGrid.addEventListener("click", (e) => {
  const shot = e.target.closest(".shot");
  // ...
});
```

**Benefits:**
- Single event listener for all images
- Works with dynamically added images
- Better performance than individual listeners

### Caption Handling

```javascript
const caption = shot.dataset.caption || img.alt;
```

Priority:
1. `data-caption` attribute on `.shot` element
2. Fallback to image `alt` text

### Body Scroll Lock

```javascript
// Lock scroll when lightbox opens

document.body.style.overflow = "hidden";

// Restore scroll when lightbox closes
function closeLightbox() {
  lightbox.classList.remove("open");
  document.body.style.overflow = "";
}
```

## Advanced Features

### Add Navigation Arrows

```javascript
let currentIndex = 0;
const images = Array.from($$('.shot img'));

function showImage(index) {
  currentIndex = index;
  const img = images[index];
  lightboxImg.src = img.src;
  lightboxCaption.textContent = img.alt;
}

function nextImage() {
  currentIndex = (currentIndex + 1) % images.length;
  showImage(currentIndex);
}

function prevImage() {
  currentIndex = (currentIndex - 1 + images.length) % images.length;
  showImage(currentIndex);
}

// Add keyboard navigation
document.addEventListener('keydown', (e) => {
  if (!lightbox.classList.contains('open')) return;
  if (e.key === 'ArrowRight') nextImage();
  if (e.key === 'ArrowLeft') prevImage();
});
```

### Dynamic Image Loading

```javascript
function addGalleryImage(src, alt, caption, rotation) {
  const figure = document.createElement('figure');
  figure.className = 'shot';
  figure.dataset.caption = caption;
  
  figure.innerHTML = `
    <div class="paper-frame" style="--rotation: ${rotation}deg;">
      <img src="${src}" alt="${alt}" loading="lazy" />
    </div>
    <div class="shot-overlay">
      <span class="shot-caption">${caption}</span>
    </div>
  `;
  
  galleryGrid.appendChild(figure);
}

// Example usage
addGalleryImage(
  'photo.jpg',
  'Festival scene',
  'Dancing performance',
  -1.2
);
```

### Stagger Animations

Reveal images with staggered entrance:

```javascript
const shots = $$('.shot');
shots.forEach((shot, i) => {
  shot.style.opacity = '0';
  shot.style.transform = 'translateY(20px)';
  shot.style.transition = `opacity 0.5s ease ${i * 0.1}s, transform 0.5s ease ${i * 0.1}s`;
  
  requestAnimationFrame(() => {
    shot.style.opacity = '1';
    shot.style.transform = 'translateY(0)';
  });
});
```
